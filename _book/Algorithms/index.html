Binary search

def binarySearch(start, end):
    if end >= start:
        mid = start + (end - start) // 2
        if arr[mid] == k:
            return mid
        if arr[mid] > k:
            return binarySearch(start, mid - 1)
        if arr[mid] < k:
            return binarySearch(mid + 1, end)
    else:
        return -1
